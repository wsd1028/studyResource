 //前端分页通用版
 function getPageData (rule) {
                    let data=rule.data;//总数据
                    let pageData = []; //分页返回数据
                    let pageIndex =  rule.pageIndex - 1;
                    let pageSize = rule.pageSize;
                    let count = 0;
                    for (let i = 0; i < data.length; i++) {
                        if (data[i + pageIndex * pageSize]) {
                            pageData.push(data[i + pageIndex * pageSize]);
                            count++;
                        } else {
                            break;
                        }
                        if (count > pageSize - 1) {
                            break;
                        }
                    }
                    return pageData;
},
